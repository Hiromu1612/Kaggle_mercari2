import matplotlib.pyplot as plt
import numpy as np

# 
models = ['RFR', 'LightGBM', 'Ridge']
rmsle_values = [0.54197, 0.54516, 0.54257]
goal_value = 0.42275

# 
colors = ['blue', 'orange', 'green']

# 
plt.figure(figsize=(8, 5))
bars = plt.bar(models, rmsle_values, color=colors)

# 
plt.axhline(y=goal_value, color='red', linestyle='--', label='Target Value (0.42275)')

# 
for bar in bars:
    yval = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, yval, round(yval, 5), ha='center', va='bottom')

# 
plt.title('RMSLE Comparison of Different Models')
plt.xlabel('Models')
plt.ylabel('RMSLE')
plt.ylim(0.4, max(rmsle_values) + 0.1)

# 
plt.legend()

# 
plt.show()

import matplotlib.pyplot as plt
import numpy as np

# 
models = ['RFR', 'LightGBM', 'Ridge']
ver1_values = [0.54197, 0.54516, 0.54257]
ver2_values = [0.56080, 0.61183, 0.73832]
goal_value = 0.42275

# X
x = np.arange(len(models))

# 
width = 0.35

# 
plt.figure(figsize=(10, 6))
bars1 = plt.bar(x - width/2, ver1_values, width, label='Version 1', color='blue')
bars2 = plt.bar(x + width/2, ver2_values, width, label='Version 2', color='orange')

# 
plt.axhline(y=goal_value, color='red', linestyle='--', label='Target Value (0.42275)')

# 
for bar in bars1:
    yval = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, yval, round(yval, 5), ha='center', va='bottom')

for bar in bars2:
    yval = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, yval, round(yval, 5), ha='center', va='bottom')

# 
plt.title('RMSLE Comparison Between Version 1 and Version 2')
plt.xlabel('Models')
plt.ylabel('RMSLE')
plt.xticks(x, models)
plt.ylim(0.4, max(ver2_values) + 0.1)

# 
plt.legend()

# 
plt.show()

import matplotlib.pyplot as plt
import numpy as np

# 
versions = ['Ver.1', 'Ver.2', 'Ver.3', 'Ver.4', 'Target']
scores = [0.54197, 0.56080, 0.46816, 0.45222, 0.42275]
ranks = [1875, 1794, 1456, 1291, 476]
top_percentages = [79, 75, 61, 54, 20]

# X
x = np.arange(len(versions))

# 
width = 0.4

# 
plt.figure(figsize=(10, 6))
bars = plt.bar(x, scores, width, color='skyblue')

# 
for i, bar in enumerate(bars):
    yval = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, yval, round(yval, 5), ha='center', va='bottom')
    plt.text(bar.get_x() + bar.get_width()/2, yval - 0.02, f'Rank: {ranks[i]}, Top: {top_percentages[i]}%', ha='center', va='top', fontsize=10)

# 
plt.title('Comparison of Versions and Target')
plt.xlabel('Versions')
plt.ylabel('Score (RMSLE)')
plt.xticks(x, versions)
plt.ylim(0.4, 0.575)

# 
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()
